## 一、需求概述

1. 根据数学函数生成的曲线创建子弹
2. 子弹沿曲线以固定速度滑动
3. 碰撞检测：子弹每帧检测是否碰到障碍物/玩家/地图边界
4. 碰到障碍物 → 爆炸，造成圆形范围伤害
5. 碰到玩家 → 玩家出局
6. 碰到地图边界 → 子弹消失
7. 到达曲线终点 → 子弹消失
8. 支持子弹轨迹预览（UI辅助）

## 三、类设计

### ShootingSystem (主类)
```csharp
// 文件: Assets/_Project/Scripts/Shooting/ShootingSystem.cs
using System.Collections.Generic;
using UnityEngine;

namespace MathGame.Shooting
{
    public class ShootingSystem : MonoBehaviour, IShootingSystem
    {
        [SerializeField] private GameConfig config;
        [SerializeField] private GameObject bulletPrefab;
        
        private ISceneManager sceneManager;
        private IPlayerManager playerManager;
        private List<BulletInfo> activeBullets = new List<BulletInfo>();
        private List<Bullet> bulletObjects = new List<Bullet>();
        
        public bool IsAnyBulletFlying => activeBullets.Count > 0;

        public void FireBullet(int ownerID, CurveData curve)
        {
            if (!curve.IsValid) return;
            
            var bullet = new BulletInfo
            {
                OwnerPlayerID = ownerID,
                Curve = curve,
                Speed = config.BulletSpeed,
                Progress = 0f,
                CurrentPosition = curve.Points[0].WorldPos,
                HasExploded = false,
                DamageRadius = playerManager.GetPlayer(ownerID).DamageRadius,
                Damage = config.BaseDamage
            };
            activeBullets.Add(bullet);

            // 实例化子弹GameObject
            var bulletObj = Instantiate(bulletPrefab, bullet.CurrentPosition, Quaternion.identity);
            var bulletComp = bulletObj.GetComponent<Bullet>();
            bulletComp.Initialize(bullet, this);
            bulletObjects.Add(bulletComp);

            GameEvent.OnBulletFired?.Invoke(bullet);
        }

        private void Update()
        {
            // 更新所有活跃子弹位置
            for (int i = activeBullets.Count - 1; i >= 0; i--)
            {
                UpdateBullet(activeBullets[i], Time.deltaTime);
            }
        }

        private void UpdateBullet(BulletInfo bullet, float dt)
        {
            if (bullet.HasExploded)
            {
                RemoveBullet(bullet);
                return;
            }

            // 1. 计算新进度
            float moveDistance = bullet.Speed * dt;
            float newProgress = MoveAlongCurve(bullet, moveDistance);
            bullet.Progress = newProgress;

            // 2. 检查是否到达终点
            if (bullet.Progress >= 1f)
            {
                RemoveBullet(bullet);
                return;
            }

            // 3. 碰撞检测
            CheckCollisions(bullet);
        }

        private float MoveAlongCurve(BulletInfo bullet, float distance)
        {
            var points = bullet.Curve.Points;
            float remainingDist = distance;
            float currentProgress = bullet.Progress;
            
            int startIdx = Mathf.FloorToInt(currentProgress * (points.Count - 1));
            startIdx = Mathf.Clamp(startIdx, 0, points.Count - 2);

            for (int i = startIdx; i < points.Count - 1; i++)
            {
                float segLength = Vector2.Distance(points[i].WorldPos, points[i + 1].WorldPos);
                if (remainingDist <= segLength)
                {
                    float t = remainingDist / segLength;
                    bullet.CurrentPosition = Vector2.Lerp(points[i].WorldPos, points[i + 1].WorldPos, t);
                    return (i + t) / (points.Count - 1);
                }
                remainingDist -= segLength;
            }

            // 到达终点
            bullet.CurrentPosition = points[points.Count - 1].WorldPos;
            return 1f;
        }

        private void CheckCollisions(BulletInfo bullet)
        {
            // 1. 地图边界检测
            if (!sceneManager.IsInsideMap(bullet.CurrentPosition))
            {
                RemoveBullet(bullet);
                return;
            }

            // 2. 障碍物碰撞检测
            foreach (var obs in sceneManager.Obstacles)
            {
                if (IsBulletHittingObstacle(bullet, obs))
                {
                    Explode(bullet, bullet.CurrentPosition);
                    return;
                }
            }

            // 3. 玩家碰撞检测
            foreach (var player in playerManager.GetAllPlayers())
            {
                if (player.PlayerID == bullet.OwnerPlayerID) continue; // 不碰自己
                if (!player.IsAlive) continue;

                if (Vector2.Distance(bullet.CurrentPosition, player.Position) < config.PlayerRadius)
                {
                    Explode(bullet, player.Position);
                    return;
                }
            }
        }

        private bool IsBulletHittingObstacle(BulletInfo bullet, ObstacleData obs)
        {
            float bulletRadius = 5f; // 子弹碰撞半径
            if (obs.Shape == ObstacleShape.Circle)
            {
                return Vector2.Distance(bullet.CurrentPosition, obs.Position) < obs.Radius + bulletRadius;
            }
            else
            {
                // 矩形碰撞：先变换到矩形局部坐标，做AABB检测
                return false; // TODO: 实现矩形碰撞
            }
        }

        private void Explode(BulletInfo bullet, Vector2 explosionCenter)
        {
            bullet.HasExploded = true;
            GameEvent.OnBulletHitObstacle?.Invoke(bullet);

            // 范围伤害计算
            var damageCalc = new DamageCalculator(config);
            damageCalc.ApplyExplosionDamage(bullet, explosionCenter, 
                                            sceneManager, playerManager);
        }

        private void RemoveBullet(BulletInfo bullet)
        {
            int idx = activeBullets.IndexOf(bullet);
            if (idx >= 0)
            {
                activeBullets.RemoveAt(idx);
                if (idx < bulletObjects.Count)
                {
                    Destroy(bulletObjects[idx].gameObject);
                    bulletObjects.RemoveAt(idx);
                }
            }
        }

        public void ClearAllBullets()
        {
            foreach (var b in bulletObjects)
                if (b != null) Destroy(b.gameObject);
            activeBullets.Clear();
            bulletObjects.Clear();
        }
    }
}
```

### CollisionDetector (碰撞检测器)
```csharp
// 文件: Assets/_Project/Scripts/Shooting/CollisionDetector.cs
namespace MathGame.Shooting
{
    /// <summary>
    /// 独立的碰撞检测工具类
    /// </summary>
    public static class CollisionDetector
    {
        /// <summary>点与圆的碰撞</summary>
        public static bool PointInCircle(Vector2 point, Vector2 center, float radius)
            => Vector2.Distance(point, center) <= radius;

        /// <summary>圆与圆的碰撞</summary>
        public static bool CircleToCircle(Vector2 c1, float r1, Vector2 c2, float r2)
            => Vector2.Distance(c1, c2) <= r1 + r2;

        /// <summary>线段与圆的碰撞</summary>
        public static bool SegmentToCircle(Vector2 a, Vector2 b, Vector2 center, float radius)
        {
            Vector2 ab = b - a;
            Vector2 ac = center - a;
            float t = Mathf.Clamp01(Vector2.Dot(ac, ab) / Vector2.Dot(ab, ab));
            Vector2 closest = a + t * ab;
            return Vector2.Distance(closest, center) <= radius;
        }

        /// <summary>点是否在AABB矩形内（矩形未旋转时）</summary>
        public static bool PointInRect(Vector2 point, Vector2 rectCenter, Vector2 size)
        {
            Vector2 half = size * 0.5f;
            return point.x >= rectCenter.x - half.x && point.x <= rectCenter.x + half.x
                && point.y >= rectCenter.y - half.y && point.y <= rectCenter.y + half.y;
        }
    }
}
```

### DamageCalculator (伤害计算器)
```csharp
// 文件: Assets/_Project/Scripts/Shooting/DamageCalculator.cs
namespace MathGame.Shooting
{
    public class DamageCalculator
    {
        private GameConfig config;

        public DamageCalculator(GameConfig config)
        {
            this.config = config;
        }

        /// <summary>应用爆炸范围伤害</summary>
        public void ApplyExplosionDamage(BulletInfo bullet, Vector2 center,
                                         ISceneManager scene, IPlayerManager players)
        {
            float radius = bullet.DamageRadius;

            // 1. 对范围内玩家造成伤害（即出局）
            foreach (var player in players.GetAllPlayers())
            {
                if (!player.IsAlive) continue;
                if (player.PlayerID == bullet.OwnerPlayerID) continue;

                if (Vector2.Distance(center, player.Position) <= radius)
                {
                    players.EliminatePlayer(player.PlayerID);
                    GameEvent.OnBulletHitPlayer?.Invoke(player.PlayerID, bullet.OwnerPlayerID);
                }
            }

            // 2. 对范围内障碍物造成伤害（如果可摧毁）
            foreach (var obs in scene.Obstacles)
            {
                if (!obs.IsDestructible) continue;
                // 检测障碍物是否在爆炸范围内
                float dist = Vector2.Distance(center, obs.Position);
                if (dist <= radius + obs.Radius)
                {
                    float damage = bullet.Damage * (1f - dist / (radius + obs.Radius)); // 衰减
                    scene.DamageObstacle(obs.ObstacleID, damage);
                }
            }

            // 3. 视觉特效（通过事件通知UI层）
            GameEvent.OnShowMessage?.Invoke($"爆炸！伤害半径: {radius:F1}");
        }
    }
}
```

## 四、子弹飞行算法

```
UpdateBullet(bullet, dt):
  moveDist = bullet.Speed * dt
  沿曲线采样点前进 moveDist 距离
  更新 bullet.Progress (0~1)
  如果 Progress >= 1: 子弹到达终点，移除
  否则: 检测碰撞
```

### 沿曲线移动
```
曲线采样点 P[0], P[1], ..., P[N-1]
bullet.Progress = t (0~1)
当前所在段 = floor(t * N)
在该段中线性插值
```
