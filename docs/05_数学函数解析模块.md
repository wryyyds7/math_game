# 05 - 数学函数解析模块 (Agent-3)

> **依赖**: 02_核心数据结构, 03_场景管理模块（只读地图尺寸）
> **被依赖**: 06_射击系统模块, 09_AI模块(V2)
> **V1交付**: 表达式解析 + 曲线生成 + 合法性验证

---

## 一、需求概述

1. 接收玩家输入的数学表达式（如 `y=x+2`, `y=sin(x)`, `y=2x^2+3x-1`）
2. 以玩家位置为原点，玩家朝向为x轴，生成2D曲线
3. 验证曲线：不超过地图边界
4. 输出 CurveData（含采样点列表和总长度）
5. 支持所有初等函数

## 四、类设计

### MathParser (主类)
```csharp
// 文件: Assets/_Project/Scripts/MathParser/MathParser.cs
using System;
using System.Collections.Generic;
using System.Globalization;
using UnityEngine;

namespace MathGame.MathParser
{
    public class MathParser : MonoBehaviour, IMathParser
    {
        [SerializeField] private GameConfig config;
        private FunctionEvaluator evaluator;
        private CurveGenerator curveGen;
        private FunctionValidator validator;

        private void Awake()
        {
            evaluator = new FunctionEvaluator();
            curveGen = new CurveGenerator(config);
            validator = new FunctionValidator();
        }

        public CurveData ParseAndGenerate(string expression, Vector2 origin, 
                                          float forwardAngle, float maxX)
        {
            var curve = new CurveData { RawExpression = expression };

            // 1. 验证语法
            if (!ValidateExpression(expression, out string error))
            {
                curve.IsValid = false;
                curve.ErrorMessage = error;
                return curve;
            }

            // 2. 生成本地采样点（局部坐标系）
            var localPoints = curveGen.GenerateLocalPoints(curveGen.ParseExpression(expression), maxX);
            
            // 3. 转换成世界坐标
            curve.Points = curveGen.ConvertToWorld(localPoints, origin, forwardAngle);

            // 4. 计算曲线长度
            curve.TotalLength = CalculateCurveLength(curve.Points);

            // 5. 验证曲线在地图内
            curve.IsValid = validator.ValidateInMap(curve.Points, config.MapWidth, config.MapHeight);
            if (!curve.IsValid)
                curve.ErrorMessage = "曲线超出地图边界";

            return curve;
        }

        public bool ValidateExpression(string expr, out string error)
        {
            return validator.ValidateSyntax(expr, out error);
        }

        public string[] GetSupportedFunctions()
        {
            return new[] { "sin", "cos", "tan", "asin", "acos", "atan",
                          "abs", "sqrt", "ln", "log", "exp", "floor", "ceil" };
        }

        public float CalculateCurveLength(List<CurvePoint> points)
        {
            float length = 0;
            for (int i = 1; i < points.Count; i++)
                length += Vector2.Distance(points[i - 1].WorldPos, points[i].WorldPos);
            return length;
        }
    }
}
```

### CurveGenerator (曲线采样生成器)
```csharp
// 文件: Assets/_Project/Scripts/MathParser/CurveGenerator.cs
namespace MathGame.MathParser
{
    public class CurveGenerator
    {
        private GameConfig config;

        public CurveGenerator(GameConfig config)
        {
            this.config = config;
        }

        /// <summary>生成本地坐标采样点</summary>
        public List<CurvePoint> GenerateLocalPoints(Func<float, float> func, float maxX)
        {
            var points = new List<CurvePoint>();
            float step = config.CurveSampleStep; // 默认5px

            for (float x = step; x <= maxX; x += step)
            {
                float y = func(x);
                points.Add(new CurvePoint { LocalX = x, LocalY = y });
            }
            // 确保包含 x=0 起点
            points.Insert(0, new CurvePoint { LocalX = 0, LocalY = func(0) });
            return points;
        }

        /// <summary>局部坐标 → 世界坐标 转换</summary>
        public List<CurvePoint> ConvertToWorld(List<CurvePoint> localPoints, 
                                                Vector2 origin, float angleDeg)
        {
            float rad = angleDeg * Mathf.Deg2Rad;
            Vector2 forward = new Vector2(Mathf.Cos(rad), Mathf.Sin(rad));
            Vector2 side = new Vector2(-Mathf.Sin(rad), Mathf.Cos(rad));

            foreach (var p in localPoints)
            {
                p.WorldPos = origin + p.LocalX * forward + p.LocalY * side;
            }
            return localPoints;
        }
    }
}
```

// 文件: Assets/_Project/Scripts/MathParser/FunctionValidator.cs
namespace MathGame.MathParser
{
    public class FunctionValidator
    {
        /// <summary>验证表达式语法</summary>
        public bool ValidateSyntax(string expression, out string error)
        {
            error = null;
            if (string.IsNullOrWhiteSpace(expression))
            {
                error = "表达式不能为空";
                return false;
            }

            // 必须以 y= 开头
            if (!expression.TrimStart().StartsWith("y="))
            {
                error = "表达式必须以 'y=' 开头";
                return false;
            }

            // 不能包含危险字符（安全限制）
            string unsafePattern = @"[^a-zA-Z0-9\+\-\*/\^%\(\)\.\,\s\=]";
            if (Regex.IsMatch(expression, unsafePattern))
            {
                error = "表达式包含不支持的字符";
                return false;
            }

            // 括号匹配
            if (!AreBracketsBalanced(expression))
            {
                error = "括号不匹配";
                return false;
            }

            return true;
        }

        /// <summary>验证曲线所有世界坐标点都在地图内</summary>
        public bool ValidateInMap(List<CurvePoint> points, float mapW, float mapH)
        {
            foreach (var p in points)
            {
                if (p.WorldPos.x < 0 || p.WorldPos.x > mapW ||
                    p.WorldPos.y < 0 || p.WorldPos.y > mapH)
                    return false;
            }
            return true;
        }

## 五、表达式解析器实现细节

### 递归下降解析器结构

```
expression → term (('+'|'-') term)*
term       → factor (('*'|'/'|'%') factor)*
factor     → power ('^' power)*
power      → ('+'|'-')? atom
atom       → NUMBER | 'x' | CONSTANT | FUNC '(' expression ')' | '(' expression ')'
FUNC       → 'sin' | 'cos' | 'tan' | 'asin' | 'acos' | 'atan' 
           | 'abs' | 'sqrt' | 'ln' | 'log' | 'exp' | 'floor' | 'ceil'
CONSTANT   → 'pi' | 'e'
NUMBER     → [0-9]+ ('.' [0-9]+)?
```

### 运算符优先级
| 优先级 | 运算符 |
|--------|--------|
| 1 (最高) | 括号、函数 |
| 2 | 一元正负 `+x`, `-x` |
| 3 | 幂 `^` |
| 4 | 乘除取模 `*`, `/`, `%` |
| 5 | 加减 `+`, `-` |
