## 一、需求概述

1. 管理游戏整体流程：初始化 → 玩家回合 → 动作结算 → 回合结束 → 检查胜负
2. 回合制：每个玩家每回合三选一（射击/闪现/旋转180度）
3. 回合推进：当前玩家行动完成后自动切换到下一存活玩家
4. 胜负判定：只剩1个存活玩家时游戏结束
5. 游戏终止条件：某玩家被淘汰 / 回合数上限（防止无限循环）

```csharp
using MathGame.Core.Data;

namespace MathGame.Core.Interfaces
{
    public interface ITurnManager
    {
        /// <summary>开始新游戏</summary>
        void StartGame(int playerCount);

        /// <summary>当前游戏阶段</summary>
        GamePhase CurrentPhase { get; }

        void SkipCurrentPlayer();

        /// <summary>获取当前行动玩家ID</summary>
        int GetCurrentPlayerID();

        /// <summary>获取回合剩余时间（秒），-1表示无限制</summary>
        float GetTurnTimeRemaining();
    }
}
```

## 三、类设计

### TurnManager (主类)
```csharp
// 文件: Assets/_Project/Scripts/TurnManagement/TurnManager.cs
using System.Collections;
using UnityEngine;

namespace MathGame.TurnManagement
{
    public class TurnManager : MonoBehaviour, ITurnManager
    {
        [SerializeField] private GameConfig config;
        [SerializeField] private float turnTimeLimit = 60f; // 每回合时限（秒），-1不限时

        private ISceneManager sceneManager;
        private IPlayerManager playerManager;
        private IShootingSystem shootingSystem;
        private TurnStateMachine stateMachine;
        private ActionResolver actionResolver;

        private int currentPlayerIndex = 0;
        public int CurrentTurnNumber { get; private set; } = 0;
        public GamePhase CurrentPhase { get; private set; } = GamePhase.WaitingForPlayers;
        private float turnTimer = 0f;

        private void Awake()
        {
            stateMachine = new TurnStateMachine();
            actionResolver = new ActionResolver();
        }

        public void StartGame(int playerCount)
        {
            StartCoroutine(GameFlow(playerCount));
        }

        private IEnumerator GameFlow(int playerCount)
        {
            // === 阶段1: 生成地图 ===
            SetPhase(GamePhase.GeneratingMap);
            sceneManager.GenerateMap();
            yield return null; // 等一帧让场景加载

            // === 阶段2: 生成玩家 ===
            SetPhase(GamePhase.WaitingForPlayers);
            for (int i = 0; i < playerCount; i++)
            {
                playerManager.SpawnPlayer($"玩家{i + 1}");
            }

            // === 阶段3: 主游戏循环 ===
            while (playerManager.GetAlivePlayerCount() > 1)
            {
                CurrentTurnNumber++;

                // 遍历每个存活的玩家
                for (int i = 0; i < playerManager.GetAllPlayers().Count; i++)
                {
                    var player = playerManager.GetAllPlayers()[i];
                    if (!player.IsAlive) continue;

                    currentPlayerIndex = i;
                    SetPhase(GamePhase.PlayerTurn);
                    GameEvent.OnTurnStarted?.Invoke(CurrentTurnNumber);

                    // 等待玩家提交动作
                    turnTimer = turnTimeLimit;
                    while (!player.HasActedThisTurn)
                    {
                        turnTimer -= Time.deltaTime;
                        if (turnTimeLimit > 0 && turnTimer <= 0)
                        {
                            SkipCurrentPlayer();
                            break;
                        }
                        yield return null;
                    }

                    // 结算动作
                    SetPhase(GamePhase.ResolvingAction);
                    yield return actionResolver.ResolveAction(player, playerManager, shootingSystem);

                    // 等待所有子弹飞行完成
                    while (shootingSystem.IsAnyBulletFlying)
                        yield return null;

                    player.HasActedThisTurn = false;
                    SetPhase(GamePhase.TurnEnd);
                    GameEvent.OnTurnEnded?.Invoke(CurrentTurnNumber);
                }
            }

            // === 阶段4: 游戏结束 ===
            SetPhase(GamePhase.GameOver);
            var winner = playerManager.GetAllPlayers().Find(p => p.IsAlive);
            GameEvent.OnGameOver?.Invoke(winner?.PlayerID ?? -1);
        }

        public void SubmitAction(int playerID, TurnActionType action, object actionData)
        {
            var player = playerManager.GetPlayer(playerID);
            if (player == null || player.PlayerID != GetCurrentPlayerID()) return;
            if (player.HasActedThisTurn) return;

            player.HasActedThisTurn = true;
            actionResolver.QueueAction(playerID, action, actionData);
            GameEvent.OnPlayerActionSelected?.Invoke(playerID, action);
        }

        public void SkipCurrentPlayer()
        {
            int id = GetCurrentPlayerID();
            var player = playerManager.GetPlayer(id);
            if (player != null) player.HasActedThisTurn = true;
            // 不执行任何动作，直接跳过
            GameEvent.OnShowMessage?.Invoke($"{player?.PlayerName} 超时，跳过回合");
        }

        public int GetCurrentPlayerID()
        {
            var player = playerManager.GetCurrentPlayer();
            return player?.PlayerID ?? -1;
        }

        public float GetTurnTimeRemaining() => turnTimer;

        private void SetPhase(GamePhase phase)
        {
            CurrentPhase = phase;
            GameEvent.OnPhaseChanged?.Invoke(phase);
        }
    }
}
```

### TurnStateMachine (状态机)
```csharp
// 文件: Assets/_Project/Scripts/TurnManagement/TurnStateMachine.cs
namespace MathGame.TurnManagement
{
    /// <summary>
    /// 回合状态机 — 定义状态转换规则
    /// </summary>
    public class TurnStateMachine
    {
        public GamePhase CurrentState { get; private set; }

        private static readonly HashSet<(GamePhase, GamePhase)> ValidTransitions = new()
        {
            (GamePhase.WaitingForPlayers, GamePhase.GeneratingMap),
            (GamePhase.GeneratingMap,     GamePhase.WaitingForPlayers),
            (GamePhase.WaitingForPlayers, GamePhase.PlayerTurn),
            (GamePhase.PlayerTurn,        GamePhase.ResolvingAction),
            (GamePhase.ResolvingAction,   GamePhase.TurnEnd),
            (GamePhase.TurnEnd,           GamePhase.PlayerTurn),  // 下一玩家
            (GamePhase.TurnEnd,           GamePhase.GameOver),    // 游戏结束
        };

        public bool CanTransition(GamePhase from, GamePhase to)
            => ValidTransitions.Contains((from, to));

        public void SetState(GamePhase newState)
        {
            if (CanTransition(CurrentState, newState))
                CurrentState = newState;
            else
                Debug.LogWarning($"非法状态转换: {CurrentState} → {newState}");
        }
    }
}
```

### ActionResolver (动作结算器)
```csharp
// 文件: Assets/_Project/Scripts/TurnManagement/ActionResolver.cs
using System.Collections;
using MathGame.Core.Data;

namespace MathGame.TurnManagement
{
    public class ActionResolver
    {
        private readonly Queue<(int playerID, TurnActionType action, object data)> actionQueue 
            = new();

        public void QueueAction(int playerID, TurnActionType action, object data)
        {
            actionQueue.Enqueue((playerID, action, data));
        }

        /// <summary>逐帧结算队列中的动作</summary>
        public IEnumerator ResolveAction(PlayerState player, IPlayerManager pm, IShootingSystem ss)
        {
            while (actionQueue.Count > 0)
            {
                var (pid, action, data) = actionQueue.Dequeue();

                switch (action)
                {
                    case TurnActionType.Shoot:
                        // data = 表达式字符串
                        string expression = data as string;
                        // 射击由ShootingSystem处理，这里只是记录
                        // 实际操作在SubmitAction中已经触发了FireBullet
                        break;

                    case TurnActionType.Blink:
                        // data = Vector2 目标位置
                        Vector2 target = (Vector2)data;
                        pm.TryBlink(pid, target);
                        break;

                    case TurnActionType.Rotate180:
                        pm.RotatePlayer(pid);
                        break;
                }
                yield return null;
            }
        }
    }
}
```

## 四、回合流程时序图

```
游戏开始
  │
  ├─ 生成地图
  ├─ 生成玩家
  │
  ├─ 回合1 ───────────────────────────────────────┐
  │   ├─ 玩家1 回合                                │
  │   │   ├─ UI显示动作选择                          │
  │   │   ├─ 玩家提交动作                            │
  │   │   ├─ 结算动作（射击→子弹飞行→碰撞→伤害）      │
  │   │   └─ 等待子弹飞行结束                        │
  │   ├─ 玩家2 回合 (同上)                          │
  │   └─ 检查胜负                                  │
  │                                                │
  ├─ 回合2 ────────────────────────────────────────┤
  │   ... (同上循环)                                │
  │                                                │
  └─ 存活玩家=1 → 游戏结束                          │
```

## 五、关键设计决策

1. **动作提交即锁定**：SubmitAction后 `HasActedThisTurn = true`，不可更改
2. **子弹飞完才切换回合**：确保回合原子性，避免并发问题
3. **超时跳过**：防止玩家无限等待，需UI倒计时
4. **事件驱动**：所有状态变化通过 GameEvent 广播，UI和音效自行监听
