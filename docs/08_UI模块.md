```csharp
// 文件: Assets/_Project/Scripts/UI/UIManager.cs
using UnityEngine;
using UnityEngine.UI;
using TMPro;

namespace MathGame.UI
{
    public class UIManager : MonoBehaviour, IUIManager
    {
        [Header("面板引用")]
        [SerializeField] private GameObject mainMenuPanel;
        [SerializeField] private GameObject gameHUDPanel;
        [SerializeField] private GameObject mathInputPanel;
        [SerializeField] private GameObject gameOverPanel;

        [Header("HUD子组件")]
        [SerializeField] private TMP_Text turnText;
        [SerializeField] private TMP_Text currentPlayerText;
        [SerializeField] private TMP_Text countdownText;
        [SerializeField] private TMP_Text playerLevelText;
        [SerializeField] private Slider expBar;
        [SerializeField] private Image[] blinkIndicators;

        [Header("数学输入")]
        [SerializeField] private TMP_InputField mathInputField;
        [SerializeField] private TMP_Text errorText;
        [SerializeField] private TMP_Text supportedFuncsText;
        [SerializeField] private Button previewButton;
        [SerializeField] private Button confirmShootButton;
        [SerializeField] private Button cancelButton;

        [Header("消息提示")]
        [SerializeField] private TMP_Text messageText;
        [SerializeField] private float messageFadeTime = 2f;

        [Header("曲线预览")]
        [SerializeField] private LineRenderer curvePreviewLine;

        // 事件
        public event Action OnStartGameClicked;
        public event Action<string> OnShootSubmitted;
        public event Action<Vector2> OnBlinkSubmitted;
        public event Action OnRotateSubmitted;

        private IMathParser mathParser;

        private void Start()
        {
            supportedFuncsText.text = "支持: sin, cos, tan, sqrt, log, ln, abs, ^, +, -, *, /";
            
            previewButton.onClick.AddListener(OnPreviewClicked);
            confirmShootButton.onClick.AddListener(OnConfirmShoot);
            cancelButton.onClick.AddListener(HideMathInputPanel);
        }

        public void ShowMainMenu()
        {
            mainMenuPanel.SetActive(true);
            gameHUDPanel.SetActive(false);
            mathInputPanel.SetActive(false);
            gameOverPanel.SetActive(false);
        }

        public void ShowGameHUD()
        {
            mainMenuPanel.SetActive(false);
            gameHUDPanel.SetActive(true);
            mathInputPanel.SetActive(false);
            gameOverPanel.SetActive(false);
        }

        public void RefreshHUD(PlayerState player, int turnNumber)
        {
            currentPlayerText.text = $"当前: {player.PlayerName}";
            turnText.text = $"回合 {turnNumber}";
            playerLevelText.text = $"Lv.{player.Level}";
            expBar.value = (float)player.CurrentExp / player.ExpToNextLevel;
            for (int i = 0; i < blinkIndicators.Length; i++)
            {
                blinkIndicators[i].color = i < player.BlinkCharges ? Color.white : Color.gray;
            }
        }

        public void ShowActionPanel() { /* 展开动作选择 */ }
        public void HideActionPanel() { /* 收起 */ }

        public void ShowMathInputPanel()
        {
            mathInputPanel.SetActive(true);
            mathInputField.text = "";
            errorText.gameObject.SetActive(false);
            curvePreviewLine.positionCount = 0;
        }

        public void HideMathInputPanel()
        {
            mathInputPanel.SetActive(false);
            curvePreviewLine.positionCount = 0;
        }

        private void OnPreviewClicked()
        {
            string expr = mathInputField.text;
            if (mathParser.ValidateExpression(expr, out string error))
            {
                errorText.gameObject.SetActive(false);
                // 生成预览曲线（需要当前玩家信息）
                OnShootSubmitted?.Invoke(expr); // 临时，预览逻辑待完善
            }
            else
            {
                errorText.text = error;
                errorText.gameObject.SetActive(true);
            }
        }

        private void OnConfirmShoot()
        {
            string expr = mathInputField.text;
            OnShootSubmitted?.Invoke(expr);
            HideMathInputPanel();
        }

        public void UpdateCurvePreview(CurveData curve)
        {
            if (curve == null || curve.Points == null) return;
            curvePreviewLine.positionCount = curve.Points.Count;
            for (int i = 0; i < curve.Points.Count; i++)
            {
                curvePreviewLine.SetPosition(i, curve.Points[i].WorldPos);
            }
        }

        public void ShowMessage(string msg, float duration = 2f)
        {
            messageText.text = msg;
            messageText.gameObject.SetActive(true);
            StartCoroutine(FadeMessage(duration));
        }

        private IEnumerator FadeMessage(float duration)
        {
            yield return new WaitForSeconds(duration);
            messageText.gameObject.SetActive(false);
        }

        public void ShowGameOver(int winnerID, string winnerName)
        {
            gameOverPanel.SetActive(true);
            gameOverPanel.GetComponentInChildren<TMP_Text>().text = $"{winnerName} 获胜！";
        }

        // === 动作按钮回调 ===
        public void OnShootButtonClick() => ShowMathInputPanel();
        public void OnBlinkButtonClick() => OnBlinkSubmitted?.Invoke(Vector2.zero); // 位置由地图点击决定
        public void OnRotateButtonClick() => OnRotateSubmitted?.Invoke();
    }
}
```

### HUDController (HUD详细控制)
```csharp
// 文件: Assets/_Project/Scripts/UI/HUDController.cs
namespace MathGame.UI
{
    /// <summary>
    /// HUD子控制器，管理倒计时、经验条动画等
    /// </summary>
    public class HUDController : MonoBehaviour
    {
        [SerializeField] private TMP_Text countdownText;
        [SerializeField] private Slider expBar;
        [SerializeField] private Image[] blinkLights;

        private ITurnManager turnManager;
        private IPlayerManager playerManager;

        /// <summary>每帧更新倒计时</summary>
        private void Update()
        {
            if (turnManager?.CurrentPhase != GamePhase.PlayerTurn) return;

            float remaining = turnManager.GetTurnTimeRemaining();
            if (remaining > 0)
            {
                countdownText.text = $"{Mathf.CeilToInt(remaining)}s";
                countdownText.color = remaining <= 10 ? Color.red : Color.white;
            }
        }

        /// <summary>刷新所有HUD元素</summary>
        public void RefreshAll()
        {
            var player = playerManager.GetCurrentPlayer();
            if (player == null) return;

            expBar.value = (float)player.CurrentExp / player.ExpToNextLevel;

            for (int i = 0; i < blinkLights.Length; i++)
                blinkLights[i].enabled = i < player.BlinkCharges;
        }
    }
}
```

### MathInputPanel (数学输入面板)
```csharp
// 文件: Assets/_Project/Scripts/UI/MathInputPanel.cs
namespace MathGame.UI
{
    /// <summary>
    /// 数学输入面板 — 支持实时语法高亮和预览
    /// </summary>
    public class MathInputPanel : MonoBehaviour
    {
        [SerializeField] private TMP_InputField inputField;
        [SerializeField] private TMP_Text errorLabel;
        [SerializeField] private TMP_Text hintText;
        [SerializeField] private Button previewBtn;
        [SerializeField] private Button confirmBtn;
        [SerializeField] private Button cancelBtn;

        public event Action<string> OnConfirmed;
        public event Action<string> OnPreviewRequested;
        public event Action OnCancelled;

        private void OnEnable()
        {
            inputField.text = "y=";       // 预设前缀
            inputField.caretPosition = 2;  // 光标定位到y=后面
            inputField.Select();
            inputField.onValueChanged.AddListener(OnInputChanged);
        }

        private void OnInputChanged(string text)
        {
            // 确保以 y= 开头
            if (!text.TrimStart().StartsWith("y="))
            {
                inputField.SetTextWithoutNotify("y=");
                inputField.caretPosition = 2;
            }

            // 实时语法高亮（可选）
        }

        public void ShowError(string msg)
        {
            errorLabel.text = msg;
            errorLabel.gameObject.SetActive(true);
        }

        public void ClearError()
        {
            errorLabel.gameObject.SetActive(false);
        }
    }
}
```
