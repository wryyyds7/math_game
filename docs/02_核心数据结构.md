### Vector2Data — 可序列化的Vector2
```csharp
// 文件: Assets/_Project/Scripts/Core/Data/Vector2Data.cs
using System;
using UnityEngine;

namespace MathGame.Core.Data
{
    /// <summary>
    /// 可序列化的二维向量，用于网络传输和JSON序列化
    /// </summary>
    [Serializable]
    public struct Vector2Data
    {
        public float x;
        public float y;

        public Vector2Data(float x, float y)
        {
            this.x = x;
            this.y = y;
        }

        public static implicit operator Vector2(Vector2Data d) => new Vector2(d.x, d.y);
        public static implicit operator Vector2Data(Vector2 v) => new Vector2Data(v.x, v.y);
    }
}
```

### PlayerState — 玩家状态
```csharp
// 文件: Assets/_Project/Scripts/Core/Data/PlayerState.cs
using System;
using UnityEngine;

namespace MathGame.Core.Data
{
    [Serializable]
    public class PlayerState
    {
        // === 基础信息 ===
        public int PlayerID;                    // 玩家唯一ID
        public string PlayerName;               // 玩家名称
        public bool IsAlive = true;             // 是否存活
        public bool IsAI;                       // 是否为AI (V2)

        // === 位置与朝向 ===
        public Vector2 Position;                // 世界坐标位置
        public float Rotation;                  // 朝向角度（度数，0=右，90=上）
        public Vector2 Forward => new Vector2(
            Mathf.Cos(Rotation * Mathf.Deg2Rad),
            Mathf.Sin(Rotation * Mathf.Deg2Rad)
        );

        // === 等级与经验 ===
        public int Level = 1;                   // 当前等级 (1~20)
        public int CurrentExp = 0;              // 当前经验值
        public int ExpToNextLevel => LevelConfig.GetExpForLevel(Level + 1);

        // === 闪现 ===
        public int BlinkCharges = 2;            // 剩余闪现次数
        public int MaxBlinkCharges = 2;         // 最大闪现次数

        // === 回合状态 ===
        public bool HasActedThisTurn = false;   // 本回合是否已行动

        // === 统计 ===
        public int Kills = 0;                   // 击杀数
        public int StarsCollected = 0;          // 收集星星数

        /// <summary>伤害半径 = 基础半径 + 等级 * 系数</summary>
        public float DamageRadius => 0.5f + Level * 0.15f;

        /// <summary>闪现范围 = 地图尺寸 * 30%</summary>
        public float BlinkRange(float mapWidth, float mapHeight)
        {
            float mapDiagonal = Mathf.Sqrt(mapWidth * mapWidth + mapHeight * mapHeight);
            return mapDiagonal * 0.3f;
        }
    }
}
```

### LevelConfig — 等级经验配置（静态计算）
```csharp
// 文件: Assets/_Project/Scripts/Core/Data/LevelConfig.cs
using UnityEngine;

namespace MathGame.Core.Data
{
    /// <summary>
    /// 等级经验值计算器
    /// 公式: 升到第N级所需总经验 = BaseExp * (N-1)^power
    ///   N=1: 0
    ///   N=2: 100 * 1^1.5 = 100
    ///   N=3: 100 * 2^1.5 ≈ 283
    ///   N=4: 100 * 3^1.5 ≈ 520
    /// </summary>
    public static class LevelConfig
    {
        private const float BaseExp = 100f;
        private const float Power = 1.5f;
        private const int MaxLevel = 20;

        /// <summary>获取升到指定等级所需的总经验</summary>
        public static int GetExpForLevel(int level)
        {
            if (level <= 1) return 0;
            return Mathf.RoundToInt(BaseExp * Mathf.Pow(level - 1, Power));
        }

        /// <summary>获取从当前等级升到下一级还需要的经验</summary>
        public static int GetExpToNextLevel(int currentLevel)
        {
            return GetExpForLevel(currentLevel + 1) - GetExpForLevel(currentLevel);
        }

        /// <summary>添加经验值，返回是否升级和新等级</summary>
        public static (bool leveledUp, int newLevel, int remainingExp) 
            AddExp(int currentLevel, int currentExp, int expToAdd)
        {
            if (currentLevel >= MaxLevel) return (false, MaxLevel, currentExp);

            int newExp = currentExp + expToAdd;
            int newLevel = currentLevel;
            
            while (newLevel < MaxLevel && newExp >= GetExpForLevel(newLevel + 1))
            {
                newLevel++;
            }
            
            return (newLevel > currentLevel, newLevel, newExp);
        }
    }
}
```

// 文件: Assets/_Project/Scripts/Core/Data/GameEvent.cs
using System;
using UnityEngine;
using UnityEngine.Events;

namespace MathGame.Core.Data
{
    /// <summary>
    /// 全局事件总线 — 所有模块通过事件通信，避免直接依赖
    /// 使用方式：GameEvent.OnPhaseChanged?.Invoke(newPhase);
    ///           GameEvent.OnPhaseChanged += HandlePhaseChanged;
    /// </summary>
    public static class GameEvent
    {
        // === 游戏流程 ===
        public static event Action<GamePhase> OnPhaseChanged;
        public static event Action<int> OnTurnStarted;          // int = 回合编号
        public static event Action<int> OnTurnEnded;            // int = 回合编号
        public static event Action<int> OnGameOver;             // int = 获胜者PlayerID

        // === 玩家事件 ===
        public static event Action<PlayerState> OnPlayerSpawned;
        public static event Action<int> OnPlayerEliminated;     // int = PlayerID
        public static event Action<int, int> OnPlayerLevelUp;   // int=PlayerID, int=新等级
        public static event Action<int, int> OnPlayerExpChanged;// int=PlayerID, int=新经验

        // === 回合动作 ===
        public static event Action<int, TurnActionType> OnPlayerActionSelected;// int=PlayerID
        public static event Action<BulletInfo> OnBulletFired;
        public static event Action<BulletInfo> OnBulletHitObstacle;
        public static event Action<int, int> OnBulletHitPlayer; // int=targetPlayerID, int=sourcePlayerID

        // === 道具 ===
        public static event Action<int, StarData> OnStarCollected; // int=PlayerID

        // === UI事件 ===
        public static event Action<string> OnShowMessage;       // 顶部消息提示
        public static event Action<BulletInfo> OnRequestBulletPreview; // 子弹轨迹预览

        // === 清理（场景切换时调用）===
        public static void ClearAllEvents()
        {
            OnPhaseChanged = null;
            OnTurnStarted = null;
            OnTurnEnded = null;
            OnGameOver = null;
            OnPlayerSpawned = null;
            OnPlayerEliminated = null;
            OnPlayerLevelUp = null;
            OnPlayerExpChanged = null;
            OnPlayerActionSelected = null;
            OnBulletFired = null;
            OnBulletHitObstacle = null;
            OnBulletHitPlayer = null;
            OnPlayerBlinked = null;
            OnStarCollected = null;
            OnShowMessage = null;
            OnRequestBulletPreview = null;
        }
    }
}

## 六、地图坐标约定

- **世界原点**: 地图左下角为 (0, 0)，右上角为 (MapWidth, MapHeight)
- **玩家朝向**: 0° = 正右方(x+)，90° = 正上方(y+)，180° = 正左方(x-)，270° = 正下方(y-)
- **曲线局部坐标系**: 原点=玩家位置，x轴正方向=玩家朝向，y轴正方向=朝向逆时针旋转90°
- **世界坐标转换公式**:
  ```
  worldPos = playerPos + localX * forward + localY * side
  forward = (cos(θ), sin(θ))
  side = (-sin(θ), cos(θ))
  ```
