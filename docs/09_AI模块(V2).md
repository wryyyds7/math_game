## 一、需求概述

在V1基础上增加AI对手，支持三种难度区分：
- **简单**: AI随机选择动作，命中率低
- **普通**: AI使用预设策略，有限计算
- **困难**: AI精确计算最优射击角度和曲线

```csharp
// 文件: Assets/_Project/Scripts/AI/AIModule.cs
using System;
using System.Collections;
using System.Collections.Generic;
using MathGame.Core.Data;
using MathGame.Core.Interfaces;

namespace MathGame.AI
{
    public class AIModule : MonoBehaviour, IAIModule
    {
        private IMathParser mathParser;
        private AIDecisionMaker decisionMaker;
        private CurveStrategy curveStrategy;

        public Difficulty CurrentDifficulty { get; private set; }

        public void Initialize(Difficulty difficulty)
        {
            CurrentDifficulty = difficulty;
            decisionMaker = new AIDecisionMaker(difficulty);
            curveStrategy = new CurveStrategy(mathParser);
        }

        public void Think(PlayerState ai, List<PlayerState> allPlayers,
                          List<ObstacleData> obstacles, Action<TurnActionType, object> onReady)
        {
            StartCoroutine(ThinkCoroutine(ai, allPlayers, obstacles, onReady));
        }

        /// <summary>模拟思考过程（加延迟让玩家感觉AI在"思考"）</summary>
        private IEnumerator ThinkCoroutine(PlayerState ai, List<PlayerState> allPlayers,
                                            List<ObstacleData> obstacles,
                                            Action<TurnActionType, object> onReady)
        {
            // 思考延迟 0.5~1.5秒，让体验更自然
            float thinkTime = UnityEngine.Random.Range(0.5f, 1.5f);
            yield return new WaitForSeconds(thinkTime);

            var (action, data) = decisionMaker.DecideAction(ai, allPlayers, obstacles, curveStrategy);
            onReady?.Invoke(action, data);
        }
    }
}
```

```csharp
// 文件: Assets/_Project/Scripts/AI/AIDecisionMaker.cs
namespace MathGame.AI
{
    public class AIDecisionMaker
    {
        private Difficulty difficulty;

        public AIDecisionMaker(Difficulty diff) { this.difficulty = diff; }

        public (TurnActionType action, object data) DecideAction(
            PlayerState ai, List<PlayerState> allPlayers,
            List<ObstacleData> obstacles, CurveStrategy strategy)
        {
            return difficulty switch
            {
                Difficulty.Easy   => DecideEasy(ai, allPlayers, obstacles, strategy),
                Difficulty.Normal => DecideNormal(ai, allPlayers, obstacles, strategy),
                Difficulty.Hard   => DecideHard(ai, allPlayers, obstacles, strategy),
                _ => DecideEasy(ai, allPlayers, obstacles, strategy)
            };
        }

        /// <summary>简单难度：随机选动作，射击使用简单函数 + 误差</summary>
        private (TurnActionType, object) DecideEasy(PlayerState ai, List<PlayerState> allPlayers,
                                                     List<ObstacleData> obstacles, CurveStrategy strategy)
        {
            // 50%射击，30%旋转，20%闪现（如果还有次数）
            float rand = UnityEngine.Random.value;
            
            if (rand < 0.5f)
            {
                // 随机曲线（y=x + 随机偏移）
                float slope = UnityEngine.Random.Range(0.5f, 2f);
                float offset = UnityEngine.Random.Range(-50f, 50f);
                string expr = offset >= 0 ? $"y={slope}*x+{offset}" : $"y={slope}*x{offset}";
                return (TurnActionType.Shoot, expr);
            }
            else if (rand < 0.8f)
            {
                return (TurnActionType.Rotate180, null);
            }
            else if (ai.BlinkCharges > 0)
            {
                // 随机闪现到附近位置
                float range = ai.BlinkRange(1920, 1080);
                float angle = UnityEngine.Random.Range(0f, Mathf.PI * 2);
                Vector2 target = ai.Position + new Vector2(
                    Mathf.Cos(angle) * range * UnityEngine.Random.value,
                    Mathf.Sin(angle) * range * UnityEngine.Random.value);
                return (TurnActionType.Blink, target);
            }
            else
            {
                float s = UnityEngine.Random.Range(0.5f, 2f);
                return (TurnActionType.Shoot, $"y={s}*x");
            }
        }

        /// <summary>普通难度：有限策略，选择最近敌人方向</summary>
        private (TurnActionType, object) DecideNormal(PlayerState ai, List<PlayerState> allPlayers,
                                                       List<ObstacleData> obstacles, CurveStrategy strategy)
        {
            // 找最近敌人
            PlayerState nearest = FindNearestEnemy(ai, allPlayers);
            if (nearest == null) return (TurnActionType.Shoot, "y=x");

            // 80%概率射击（如果有合适角度）
            if (UnityEngine.Random.value < 0.8f)
            {
                string expr = strategy.GenerateSimpleCurve(ai, nearest, obstacles);
                return (TurnActionType.Shoot, expr);
            }
            else if (ai.BlinkCharges > 0 && UnityEngine.Random.value < 0.5f)
            {
                // 闪现到敌人附近或远离敌人
                Vector2 dir = (nearest.Position - ai.Position).normalized;
                float range = ai.BlinkRange(1920, 1080);
                Vector2 target = UnityEngine.Random.value > 0.5f
                    ? ai.Position + dir * range * 0.8f         // 靠近
                    : ai.Position - dir * range * 0.8f;        // 远离
                return (TurnActionType.Blink, ClampToMap(target));
            }
            else
            {
                return (TurnActionType.Rotate180, null);
            }
        }

        /// <summary>困难难度：精确计算最优曲线</summary>
        private (TurnActionType, object) DecideHard(PlayerState ai, List<PlayerState> allPlayers,
                                                     List<ObstacleData> obstacles, CurveStrategy strategy)
        {
            // 1. 评估所有敌人，选最佳目标
            PlayerState bestTarget = FindBestTarget(ai, allPlayers, obstacles);
            if (bestTarget == null) return (TurnActionType.Shoot, "y=x");

            // 2. 计算最佳朝向
            float bestAngle = strategy.CalculateOptimalAngle(ai, bestTarget, obstacles);

            // 3. 如果需要调整角度很大且不能旋转到位，考虑闪现
            float angleDiff = Mathf.Abs(Mathf.DeltaAngle(ai.Rotation, bestAngle));
            
            // 不需要旋转就射击（已经朝向对）
            if (angleDiff < 30f)
            {
                string expr = strategy.GenerateOptimalCurve(ai, bestTarget, obstacles);
                return (TurnActionType.Shoot, expr);
            }

            // 需要旋转180度就能对准？（允许180度旋转）
            float afterRotate = (ai.Rotation + 180f) % 360f;
            if (Mathf.Abs(Mathf.DeltaAngle(afterRotate, bestAngle)) < 45f)
            {
                return (TurnActionType.Rotate180, null);
            }

            // 否则考虑闪现到有利位置
            if (ai.BlinkCharges > 0)
            {
                Vector2 blinkTarget = strategy.CalculateBestBlinkPosition(ai, bestTarget, obstacles);
                return (TurnActionType.Blink, blinkTarget);
            }

            // 无闪现，只能旋转
            return (TurnActionType.Rotate180, null);
        }
```

```csharp
    /// AI专用的曲线生成策略

    public class CurveStrategy
    {
        private IMathParser mathParser;

        public CurveStrategy(IMathParser parser)
        {
            this.mathParser = parser;
        }

        /// <summary>生成简单直线/抛物线（普通难度）</summary>
        public string GenerateSimpleCurve(PlayerState ai, PlayerState target,
                                          List<ObstacleData> obstacles)
        {
            // 计算瞄准敌人的最佳简单函数
            // 直线：y = k*x（调整斜率使子弹经过目标附近）
            
            Vector2 toTarget = target.Position - ai.Position;
            Vector2 forward = ai.Forward;
            Vector2 side = new Vector2(-forward.y, forward.x);

            // 目标在局部坐标系中的位置
            float localX = Vector2.Dot(toTarget, forward);
            float localY = Vector2.Dot(toTarget, side);

            if (localX <= 0) return "y=x"; // 目标在背后

            // 简单直线 y = (localY/localX) * x
            float slope = localY / localX;
            string expr = slope >= 0
                ? $"y={slope:F2}*x"
                : $"y={slope:F2}*x";
            return expr;
        }
```

## 四、AI决策流程图

```
AI回合开始
  │
  ├─ 评估自身状态（等级，闪现次数，位置）
  ├─ 扫描敌人（距离，等级，是否在视野/射程内）
  ├─ 扫描障碍物（判断是否有阻碍）
  │
  ├─ [简单] 随机动作
  │   ├─ 50% 射击（y=kx+b，随机参数+大误差）
  │   ├─ 30% 旋转180度
  │   └─ 20% 闪现（随机位置）
  │
  ├─ [普通] 有限策略
  │   ├─ 80% 射击（直线瞄最近敌人+误差）
  │   ├─ 10% 闪现（靠近/远离敌人）
  │   └─ 10% 旋转
  │
  └─ [困难] 最优决策
      ├─ 搜索360度最佳射击角度
      ├─ 评估多种曲线（直线/抛物线/正弦波）
      ├─ 如果当前角度不佳 → 旋转或闪现调整
      └─ 输出最优曲线表达式
```
