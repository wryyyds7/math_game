## 一、需求概述

1. 玩家出生点随机生成，随机朝向
2. 每局2次闪现（瞬移），闪现范围=地图对角线×30%
3. 玩家等级系统：吃星星升级，伤害半径随等级增长
4. 玩家死亡（被子弹击中或出界）
5. 生命值=1（即被击中直接出局）

## 三、类设计

### PlayerManager (主类)
```csharp
// 文件: Assets/_Project/Scripts/Player/PlayerManager.cs
namespace MathGame.Player
{
    public class PlayerManager : MonoBehaviour, IPlayerManager
    {
        [SerializeField] private GameConfig config;
        [SerializeField] private ISceneManager sceneManager;

        private List<PlayerState> players = new List<PlayerState>();
        private int nextPlayerID = 0;
        private int currentPlayerIndex = 0;

        public List<PlayerState> GetAllPlayers() => players;
        public PlayerState GetPlayer(int id) => players.Find(p => p.PlayerID == id);
        public PlayerState GetCurrentPlayer() => players[currentPlayerIndex];
        public int GetAlivePlayerCount() => players.Count(p => p.IsAlive);
        public bool IsPlayerAlive(int id) => GetPlayer(id)?.IsAlive ?? false;

        public PlayerState SpawnPlayer(string name, bool isAI)
        {
            Vector2 pos = sceneManager.GetValidSpawnPoint();
            float rot = Random.Range(0f, 360f);
            var state = new PlayerState
            {
                PlayerID = nextPlayerID++,
                PlayerName = name,
                Position = pos,
                Rotation = rot,
                IsAI = isAI,
                BlinkCharges = config.MaxBlinkCharges,
                MaxBlinkCharges = config.MaxBlinkCharges
            };
            players.Add(state);
            GameEvent.OnPlayerSpawned?.Invoke(state);
            return state;
        }

        public bool TryBlink(int playerID, Vector2 target)
        {
            var player = GetPlayer(playerID);
            if (player == null || !player.IsAlive || player.BlinkCharges <= 0)
                return false;

            float blinkRange = player.BlinkRange(config.MapWidth, config.MapHeight);
            if (Vector2.Distance(player.Position, target) > blinkRange)
                return false;
            if (!sceneManager.IsInsideMap(target))
                return false;
            // 闪现不受障碍物阻挡（设计决策）

            player.Position = target;
            player.BlinkCharges--;
            GameEvent.OnPlayerBlinked?.Invoke(playerID, target);
            return true;
        }

        public void RotatePlayer(int playerID)
        {
            var player = GetPlayer(playerID);
            if (player == null) return;
            player.Rotation = (player.Rotation + 180f) % 360f;
        }

        public (bool, int) AddExp(int playerID, int exp)
        {
            var player = GetPlayer(playerID);
            if (player == null) return (false, player.Level);

            var result = LevelConfig.AddExp(player.Level, player.CurrentExp, exp);
            player.CurrentExp = result.remainingExp;
            
            if (result.leveledUp)
            {
                player.Level = result.newLevel;
                GameEvent.OnPlayerLevelUp?.Invoke(playerID, result.newLevel);
            }
            GameEvent.OnPlayerExpChanged?.Invoke(playerID, player.CurrentExp);
            return (result.leveledUp, result.newLevel);
        }

        public void EliminatePlayer(int playerID)
        {
            var player = GetPlayer(playerID);
            if (player == null || !player.IsAlive) return;
            
            player.IsAlive = false;
            GameEvent.OnPlayerEliminated?.Invoke(playerID);
        }

        /// <summary>推进到下一个存活玩家（由TurnManager调用）</summary>
        public void AdvanceToNextPlayer()
        {
            int start = currentPlayerIndex;
            do
            {
                currentPlayerIndex = (currentPlayerIndex + 1) % players.Count;
            }
            while (!players[currentPlayerIndex].IsAlive && currentPlayerIndex != start);
        }
    }
}
```

namespace MathGame.Player
{
    /// <summary>
    /// 挂载在Player预制体上，负责视觉效果、移动动画
    /// </summary>
    public class PlayerController : MonoBehaviour
    {
        public int PlayerID;
        private PlayerState state;
        private SpriteRenderer spriteRenderer;
        private IPlayerManager playerManager;

        public void Initialize(int playerID, IPlayerManager pm)
        {
            this.PlayerID = playerID;
            this.playerManager = pm;
            this.state = pm.GetPlayer(playerID);
            UpdateVisual();
        }

        private void Update()
        {
            if (state == null) return;
            // 平滑移动到目标位置（闪现/回合移动）
            transform.position = Vector2.Lerp(transform.position, state.Position, 0.1f);
            // 平滑旋转
            float targetAngle = state.Rotation;
            transform.rotation = Quaternion.Lerp(transform.rotation,
                Quaternion.Euler(0, 0, targetAngle - 90), 0.1f); // -90因为精灵默认朝上
            UpdateVisual();
        }

        private void UpdateVisual()
        {
            // 显示等级、名字标签、生命状态等
        }
    }
}

## 四、出生点算法

```
GetValidSpawnPoint():
  DO:
    pos.x = RandomRange(config.PlayerSpawnMargin, MapWidth - config.PlayerSpawnMargin)
    pos.y = RandomRange(config.PlayerSpawnMargin, MapHeight - config.PlayerSpawnMargin)
  WHILE (IsCollidingWithObstacle(pos, config.PlayerRadius) 
         OR IsTooCloseToOtherPlayers(pos))
  RETURN pos
```

## 五、闪现实现要点

```
TryBlink(playerID, targetPos):
  1. 检查闪现次数 > 0
  2. 计算距离：|targetPos - playerPos| ≤ 地图对角线 × 30%
  3. 检查目标在地图内
  4. 不检查障碍物碰撞（闪现穿过一切）
  5. 更新PlayerState.Position = targetPos
  6. 触发OnPlayerBlinked事件
  7. 返回true
```
